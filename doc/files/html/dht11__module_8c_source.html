33/*
34 * Waits for the GPIO to change its level before reaching
35 * the timeout (in us).
36 */
37static int wait(struct GpioRegisters *pGPIORegisters, int GPIO, bool level, int timeout)
38{
39 int cnt = 0;
40 while (level == GetGPIOValue(pGPIORegisters,GPIO)) {
41 if (cnt > timeout) {
42 return 0;
43 }
44 cnt++;
45 udelay(1);
46 }
47 return 1;
48}

50ssize_t dht11_device_read(struct file *pfile, char __user *p_buff,size_t len, loff_t *poffset){
51 struct GpioRegisters *pdev;
52
53 bool lastState = 0;
54 uint8_t bitCounter = 0;
55 uint8_t timeCounter = 0;
56
57 dht11_t sensorData;
58
59 pr_alert("%s: called\n",__FUNCTION__);
60
61 if(unlikely(pfile->private_data == NULL))
62 return -EFAULT;
63
64 pdev = (struct GpioRegisters *)pfile->private_data;
65
66 SetGPIOFunction(pdev, DATA, OUTPUT); //Set pin to output
67 SetGPIOValue(pdev,DATA,LOW); //Set pin LOW
68 mdelay(18); //Wait for 18ms
69 SetGPIOValue(pdev,DATA,HIGH); //Set pin HIGH
70 udelay(40); //Wait for 40us
71 SetGPIOFunction(pdev, DATA, INPUT); //Set pin to input
72
73 if(!wait(pdev, DATA, 0,100)) //Wait for the sensor's acknowledge
74 {
75 pr_alert("%s: Timeout. No ACK from sensor.\n",__FUNCTION__,len);
76 return 0;
77 }
78
79 if(!wait(pdev, DATA, 1,100)) //Wait for the sensor's response
80 {
81 pr_alert("%s: Timeout.\n",__FUNCTION__,len);
82 return 0;
83 }
84
85 int i;
86 for(i = 0; i < 80; i++)
87 {
88 timeCounter=0;
89
90 while(GetGPIOValue(pdev,DATA) == lastState)
91 {
92 timeCounter++;
93 udelay(1);
94 if(timeCounter == 255)
95 {
96 pr_alert("%s: Sampling error.\n",__FUNCTION__,len);
97 return 0;
98 }
99 }
100 lastState = GetGPIOValue(pdev,DATA);
101
102 if( ( i >= 1) && (i % 2) ) //Only get data bits
103 {
104 sensorData.CompleteSample[bitCounter / 8] <<= 1;
105 if(timeCounter > 30)
106 sensorData.CompleteSample[bitCounter / 8] |= 1;
107 bitCounter++;
108 }
109 }
110
111 if( bitCounter >= 40)
112 {
113 //Calculate the checksum
114 u_int8_t checksum=0;
115 for(i = 0; i < 4; i++)
116 checksum += sensorData.CompleteSample[i];
117 sensorData.checksum=checksum;
118
119 if(sensorData.checksum != sensorData.CompleteSample[4]) //Verify the checksum
120 {
121 pr_alert("%s: Sampling error. Checksum mismatch: %d != %d\n",__FUNCTION__,sensorData.checksum,sensorData.CompleteSample[4]);
122 return 0;
123 }
124
125 sensorData.HumidityI = sensorData.CompleteSample[0];
126 sensorData.HumidityD = sensorData.CompleteSample[1];
127 sensorData.TemperatureI = sensorData.CompleteSample[2];
128 sensorData.TemperatureD = sensorData.CompleteSample[3];
129
130 int ret;
131 do{
132 ret=copy_to_user(p_buff,(void *)&sensorData,sizeof(sensorData)); //Send to user space
133 }while(ret);
134
135 return len;
136 }
137
138 return 0;
139}
